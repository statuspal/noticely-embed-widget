This demo shows how the banner works in production. The banner automatically initializes based on window.NoticelyWidgetConfig. 

<script>
  window.NoticelyWidgetConfig = {
    origin: 'https://status.noticely.io',
    theme: 'auto', // or 'dark', 'light'
    position: 'bottom-right', // or 'bottom-left', 'top-right', 'top-left'
    enabled: true // set to false to disable the banner
  };
</script>
